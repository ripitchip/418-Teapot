# 217. Contains Duplicate

{{#tab name="C++" }}

```cpp
using namespace std;
#include <unordered_set>
#include <vector>
class Solution {
public:
  bool containsDuplicate(vector<int> &nums) {
    unordered_set<int> seen;
    for (int i : nums) {
      if (seen.count(i)) {
        return true;
      }
      seen.insert(i);
    }
    return false;
  }
};

```

class Solution(object):
